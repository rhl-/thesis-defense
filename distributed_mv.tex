\begin{frame}[fragile]
\frametitle{derive persistence with mayer vietoris.}
\[  \left(%
  \vcenter\bgroup\hbox\bgroup
  \tikzpicture[
   scale=.6,
    x=.75\baselineskip,
    y=.75\baselineskip,
  ]
    \mblock[afragreen](0,1)\temporal<2>{\partial_{1}}{R_1 \cdot D_1}{P_1}(5,5)
     \mblock[afrablue](5,-4)\temporal<2>{\partial_{2}}{R_2 \cdot D_2}{P_2}(5,5)
     \mblock[afrapurple](10,-9)\temporal<2>{\partial_{3}}{R_3 \cdot D_3}{P_3}(5,5)
      \mblock[afragreen](15,1)(2,5)
      \mblock[afragreen](19,1)\temporal<2>{L_1}{S_1L_1}{S_1L_1}(4,5)
     % \mblock[afragreen](21,1)L_1(2,5) 
     
      \mblock[afrablue](15,-4)\temporal<2>{L_2}{S_2L_2}{S_2L_2}(4,5)
     %\mblock[afrablue](17,-4)L_2(2,5)
     \mblock[afrablue](21,-4)(2,5)
     
     \mblock[afrapurple](17,-9)\temporal<2>{L_3}{S_3L_3}{S_3L_3}(6,5)
     % \mblock[afrapurple](19,-9)L_3(2,5) 
      % \mblock[afrapurple](21,-9)L_3(2,5)
       \mblock[gray](15,-13)I(8,4)       
  \endtikzpicture
  \egroup
  \egroup \right)
   \]
   \only<1-4>{
  All processors execute these operations with no communication!
  \begin{enumerate}
  \item[Step 1]<1-> Reduce all blocks (except I) of a fixed color independently. $\partial_i = R_i \cdot D_i$ 
  \item[Step 2]<2-> Row reduce disjoint union $S^{-1}[R_i \mid L_i] \rightarrow [P_i \mid \tilde{L}_i]$ 
  \item[Step 3]<3-> Perform all \textbf{valid} columns adds from $P_i$ into $\tilde{L}_i$ (perfect parallelism). 
  \item[Sofar]<4-> Not finished yet, but, have not done anything wrong.
  \end{enumerate}
  }
  \only<5->{
  \begin{enumerate}
  \item[Sparsity:]<5-> Now each $P_i$ has at most 1 nonzero per column. 
  \item[Fill in:]<6-> If $D.U.$ has size $m$ and total size is $m+n$ then fill in is at most $O(mn)$ down from $O((m+n)^2)$
  \item[Issue:]<7-> After permuting, naive reduction, could fill in entire matrix.
  \item[Solution:]<8-> Row operations to the rescue!
  \end{enumerate}
  }
\end{frame}

\begin{frame}{Future Directions}
\begin{enumerate}
\item There is still some room to improve the space complexity of this algorithm, for example, by reducing the factor $n$ in $O(mn)$
\item Algorithm is top heavy, eventually a large matrix is on one machine. 
\item We wanted to use M.V. to avoid this! Still some room for more cleverness here.
\end{enumerate}
